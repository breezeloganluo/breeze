 35 	function _createPost(){
 36 		var _postData = [];
 37 		return {
 38 			addPost:function(__oneData){
 39 				_postData.push(__oneData);
 40 			},
 41 			clear:function(){
 42 				_postData=[];
 43 			},
 44 			doPost:function(__finishedCallback,__this){
 45 				if (_postData == null || _postData.length == 0){
 46 					return;
 47 				}
 48 				var pObj = {};
 49 				for (var i =0;i<_postData.length;i++){
 50 					//保留回调函数以及序号
 51 					pObj[i] = _postData[i].callBack;
 52 					pObj[i].thisObj = _postData[i].callbackThisObj;
 53 					delete _postData[i].callbackThisObj;
 54 					delete _postData[i].callBack;
 55 				}
 56 				var dataStr = _useAPI.lang.toJSONString(_postData);
 57 				$.post(_doserverUrl,{data:dataStr},
 58 				function(__returndata){
 59 					for (var key in __returndata){
 60 						if (!__returndata.hasOwnProperty(key)){
 61 							continue;
 62 						}
 63 						pObj[key] && pObj[key].apply(pObj[key].thisObj,[__returndata[key].code,__returndata[key].data]);
 64 					}
 65 //					for(var i=0;i<__returndata.length;i++){
 66 //						var sName = __returndata[i].name;
 67 //						pObj[sName] && pObj[sName].callBack(__returndata[i].code,__returndata[i].data);
 68 //					}
 69 					//最后所有都结束了，统一给一个通知
 70 					__finishedCallback && __finishedCallback.apply(__this||this,[]);
 71 				},
 72 				"json");
 73 			}
 74 		};

486 				continue;
487 			}

489 			var appDefObj = defObj[appName];
490 			if (appDefObj.type != null && /container/i.test(appDefObj.type)){				
491 				continue;
492 			}
493 			if (appDefObj.appClass == null){
494 				seajs.log("class defind not found:");
495 				continue;
496 			}
497 			
498 			var classObj = _gadget[appDefObj.appClass];
499 			if (classObj == null){
500 				seajs.log("class not found :"+appDefObj.appClass);
501 				return;
502 			}
503 			var param = appDefObj.param;
504 			//下面组装新的APP			
505 			var app = _app[inId] = $("#"+appName)[0].app = {};
506 			app.id = inId;
507 			
508 			app.dom = $("#"+app.id);
509 			//复制所有class
510 			for (var name in classObj){
511 				if (classObj.hasOwnProperty(name)){
512 					app[name] = classObj[name];
513 				}
514 			};
515 			//加入系统参数
516 			app.param = param;
517 			//加入每个应用的视图,所以class 为 AppView的作为视图，而且id作为视图id
518 			app.view = {};
519 			for(var viewName in appDefObj.view){
520 				if (!appDefObj.view.hasOwnProperty(viewName)){
521 					continue;
522 				}
523 				var HTMLfunction = appDefObj.view[viewName];
524 				if (!HTMLfunction){
525 					seajs.log("can not read view :"+viewName);
526 					continue;
527 				}
528 				//到时补充替换事件处理能力
529 				var inhtml = HTMLfunction();
530 				app.view[viewName] = inhtml;
531 			};
532 			//绑定工具
533 			app.API = _createFWTools(app);
534 			//创建自定函数对象
535 			app.MY = {};
536 		}

543 	*1. 页面上所有class是FWApp的都是一个具体的APP实体，或者是容器实体
544 	*2. 该标签的ID就是对应实例ID
545 	*3. 该标签有一段注释是以<!--@classname@开头的其内容是一个json的完整注释就是一个完整的该APP的参数
546 	*4. 该参数将被转换成一个appparam的成员内容
547 	*5. 该方法将原来的class和appparam一起转成一个新的实例并丢到_app里面
548 	*新增处理解析容器能力，容器是一个由一个主APP加上若干个容器特定API构成。容器API仅仅对外提供容器内部APP的现实隐藏功能。
549 	*容器采用class=FWContainer形式，容器的dom的id和主app的id相同，建立好关联后，需要将容器的id属性删除掉
550 	*/
551 	var _createInstancesByHtmlDom = function(){
552 		//先解释和处理APP的
553 		$.each($(".FWApp"),function(){
554 			var inId = this.id;			
555 			var htmlStr = $(this).html();
556 			//取里面的系统配置参数
557 			var execResult =  /<!--\s*@(\s*[^@]+)\s*@([\s\S]+?)-->/.exec(htmlStr);
558 			if (execResult == null){
559 				seajs.log("param defind not found:"+inId);
560 				return;
561 			}
562 			var classObj = _gadget[execResult[1]];
563 			if (classObj == null){
564 				seajs.log("class not found :"+execResult[1]);
565 				return;
566 			}
567 			var param = eval("(" + execResult[2] +")");
568 			//下面组装新的APP
569 			
570 			/**
571 			 * @namespace
572 			 * @author Logan 
573 			 * @name APP 
574 			 * @description 这是app对象，而在我们FW里面注册的gadget实际上用this代表了一个具体的APP
575 			 */
576 			var app = _app[inId] = this.app = {};
577 			/**
578 			 * @name id 
579 			 * @memberOf APP
580 			 * @description APP的id
581 			 */
582 			app.id = inId;
583 			/**
584 			 * @name dom
585 			 * @memberOf APP 
586 			 * @description 这个APP代表的html的dom节点，注意，不是jquery的container
587 			 */
588 			app.dom = $(this);
589 			//复制所有class
590 			for (var name in classObj){
591 				if (classObj.hasOwnProperty(name)){
592 					app[name] = classObj[name];
593 				}
594 			}
595 			/**
596 			 * @name param 
597 			 * @memberOf APP
598 			 * @description 在该APP中声明的参数
599 			 */
600 			app.param = param;
601 
602 			/**
603 			 * @name view 
604 			 * @memberOf APP
605 			 * @description 加入每个应用的视图,key是视图id，value是视图字符串
606 			 */
607 			app.view = {};
608 			$.each($(this).children(),function(){
609 				if(this.id ==''){
610 					return;
611 				}
612 				var regStr = "<"+this.tagName+"\\s+[^>]*?id=[\"']?"+this.id+"[\"']?\\s*[^>]*>";
613 				var reg=new RegExp(regStr,"i");
614 				var execResult = reg.exec(htmlStr);
615 				var head = "";
616 				var tail = "";
617 				if (execResult!= null){
618 					head = execResult[0];
619 					tail = "</"+this.tagName+">";
620 					//alert(head+tail);
621 				}
622 				//到时补充替换事件处理能力
623 				var inhtml = $(this).html();
624 				inhtml = inhtml.replace(/<!--/ig,"<!--").replace(/-->/ig,"-->");				
625 				app.view[this.id] = head+inhtml+tail;
626 			});
627 			/**
628 			 * @name API 
629 			 * @memberOf APP
630 			 * @description 应用于该APP中的每一个API函数
631 			 * @see API
632 			 */
633 			app.API = _createFWTools(app);
634 			//创建自定函数对象
635 			/**
636 			 * @name MY 
637 			 * @memberOf APP
638 			 * @description 用于扩展，即留给每个APP由用户自定义的函数或类对象
639 			 */
640 			app.MY = {};
641 			$(app.dom).html("");
642 		});
643 		//然后解析和处理容器的
644 		$.each($(".FWContainer"),function(){
645 			//记录信息并把ID改掉
646 			var idx = this.id;
647 			$(this).removeAttr("id");
648 			var mainapp = _app[idx];
649 			if (!mainapp){
650 				seajs.log("BreezeFW._createInstances:main app is not exist id is:" + idx);
651 				return;
652 			}
653 			var container = _container[idx] = mainapp.container = {};
654 			container.appArray = [];
655 			//下面收集容器里面的app信息
656 			$(this).find(".FWApp").each(function(){
657 				if (this.id == idx){
658 					//这是主APP，忽略
659 					return;
660 				}
661 				var innerApp = _app[this.id];
662 				innerApp.dom.css("display","none");//出事情况下，不显示
663 				if (!innerApp){
664 					seajs.log("BreezeFW._createInstances:app is not exist id is:" + this.id);
665 					return;
666 				}
667 				container.appArray.push(innerApp);
668 			});
669 			
670 			//下面给mainapp增加函数
671 			mainapp.API.CShow = function(id){
672 				if (/string/i.test(id)){
673 					//如果是字符串，那么就是取ID
674 					for (var i=0;i<container.appArray.length;i++){
675 						if (container.appArray[i].id == id){
676 							container.appArray[i].dom.css("display","");
677 							break;
678 						}
679 					}
680 				}else{
681 					container.appArray[id].dom.css("display","");
682 				}
683 			};
684 			mainapp.API.CShowOnly = function(id){
685 				if (/string/i.test(id)){
686 					//如果是字符串，那么就是取ID
687 					for (var i=0;i<container.appArray.length;i++){
688 						if (container.appArray[i].id == id){
689 							container.appArray[i].dom.css("display","");
690 						}else{
691 							container.appArray[i].dom.css("display","none");
692 						}
693 					}
694 				}else{
695 					for (var i=0;i<container.appArray.length;i++){
696 						if (i == id){
697 							container.appArray[i].dom.css("display","");
698 						}else{
699 							container.appArray[i].dom.css("display","none");
700 						}
701 					}					
702 				}
703 			};
704 		});
705 	};
706 	
707 	
708 });
