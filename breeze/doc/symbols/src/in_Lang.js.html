 35 	 * @param {String}pName 参数名
 36 	 * @returns 参数名对应的参数值
 37 	 */
 38 	_result.getParameter=function(__pName){
 39 		var url = window.location.toString();
 40 		var execResult = (new RegExp(__pName+"=([\\w%\\.]+)")).exec(url);
 41 		if (execResult){
 42 			return execResult[1];
 43 		}
 44 		return null;

 47 	/**
 48 	 * @function
 49 	 * @memberOf lang
 50 	 * @name save
 51 	 * @desctiption 本地存储,存
 52 	 * @param {String} name 存储名称
 53 	 * @param {Object} data 被存储的对象，如果为null，则是把name删除掉
 54 	 * @param {boolean} [isPermanent] 是否永久性存储，true是永久存储，false或默认不填是会话存储
 55 	 * @see load
 56 	 * @example
 57 	 * FW.use().save("abc",{a:'hello'});//存储
 58 	 * FW.use().save("abc",null);//删除
 59 	 * 
 60 	 */
 61 	_result.save=function(name,data,isPermanent){
 62 		if(!window.localStorage || !window.sessionStorage ){
 63 			seajs.log("browser not supprt window.loalStorage!");
 64 			return;
 65 		}
 66 		var storage = (isPermanent)?localStorage:sessionStorage;
 67 		if (data){
 68 			storage.setItem(name,APICtr.JsonAPI.toJSONString(data));					
 69 		}else{
 70 			//如果是null，表示删除
 71 			storage.removeItem(name);
 72 		}
 73 	};

 79 	 * @desctiption 本地存储，取
 80 	 * @param {String} name 存储名称
 81 	 * @param {boolean} [isPermanent] 是否永久性存储，true是永久存储，false或默认不填是会话存储
 82 	 * @return 返回本地存储name对应的值
 83 	 * @see save
 84 	 * @example
 85 	 * FW.use().load("abc");
 86 	 * 
 87 	 */
 88 	_result.load=function(name,isPermanent){
 89 		if(!window.localStorage || !window.sessionStorage ){
 90 			seajs.log("browser not supprt window.loalStorage!");
 91 			return null;
 92 		}
 93 		var storage = (isPermanent)?localStorage:sessionStorage;
 94 		var resultJson = storage.getItem(name);
 95 		return eval("("+resultJson+")");
 96 	};

167 	/**
168 	 * @function
169 	 * @memberOf lang
170 	 * @name bindFormList
171 	 * @desctiption 创建表单分页列表页
172 	 * @param {Object}_desc 对象描述符 
173 	 * @param {JqueryContainer}_dom 加载生成的表单，挂接到dom节点
174 	 * @param {String}_countService 请求列表数据总数的serviceName  查询结果获取格式data[0].count
175 	 * @param {String}_countPackage 请求列表数据总数的package
176 	 * @param {Object}_countParam 请求列表数据总数的param 
177 	 * @param {String}_qConService 请求列表内容的serviceName
178 	 * @param {String}_qConPackage 请求列表内容的package
179 	 * @param {Number}_pagesize 列表m每页显示多少条数据
180 	 * @param {function}_formListFun 获取分页按钮对应页码的真实数据函数
181 	 * @param {function}_bindFormListCallBack 分页列表显示后的回调函数，用途例如：用于给列表绑定按钮事件
182 	 */
183 	_result.bindFormList = function(_desc,_dom,_countService,_countPackage,_countParam,_qConService,_qConPackage,_pagesize,_formListFun,_bindFormListCallBack){
184 		// this.FW.blockUI("<a href=''>alec</a>",0,0,400,400,0);
185 		var _this = this;
186 
187 		//计算list的数据总数
188 		 _this.FW.doServer(_countService,_countPackage,_countParam,function(code,data){
189 			 if(code==0){
190 
191 				//储存数据总数
192 				// var dataCount = data;
193 				var dataCount = data[0].count;
194 
195 				//储存总页数
196 				var pageCount = Math.ceil(dataCount/_pagesize);
197 				
198 				//如果pageCount等于0,则显示空列表
199 				if(!pageCount){
200 					APICtr.FormAPI.createFormListByObjDesc(_desc,_dom);
201 					return;
202 				}
203 
204 				//显示列表函数
205 				function reShowFormList(pageNum){
206 
207 					//获取最新newData数据
208 					var newParam = _formListFun(pageNum);
209 
210 					_this.FW.doServer(_qConService,_qConPackage,newParam,function(code,data){
211 						if(code == 0){
212 							
213 							//生成表单								
214 							if(data && data.length){
215 								//转换data中数组字符串的情况为数组
216 								for(var v_prop in _this.contentDesc){
217 									if(_this.contentDesc[v_prop].type == 'List'){
218 										for(var i=0;i<data.length;i++){
219 											data[i][v_prop] = FW.use().evalJSON(data[i][v_prop]);
220 										}
221 									}
222 								}
223 								var newData = data;
224 								// alert(_this.FW.use().toJSONString(newData));
225 
226 								if(!newData){
227 									alert("获取数据失败！");
228 									return
229 								}
230 
231 								//重新显示列表数据
232 								APICtr.FormAPI.createFormListByObjDesc(_desc,_dom,newData);
233 								
234 								//更新数字连接
235 								strPageNum = "";
236 								
237 								var begNo = pageNum - 3 > 0 ? pageNum - 2 : 1;
238 								if(pageCount - pageNum < 2){begNo = begNo + pageCount - pageNum - 2 > 0 ? begNo + pageCount - pageNum - 2 : 1;}
239 								var endNo = begNo + 5 > pageCount ? pageCount + 1 : begNo + 5;
240 								for(var i=begNo; i<endNo; i++){	
241 									if(pageNum == i){
242 										strPageNum += '<li class="pagenum active"><a href="javascript:void(0);">'+i+'</a></li>\n';
243 									}else{
244 										strPageNum += '<li class="pagenum"><a href="javascript:void(0);">'+i+'</a></li>\n';
245 									}
246 								}
247 
248 								//重置分页列表的dom节点
249 								var strPageBtn = "<div class='dataTables_paginate paging_bootstrap pagination'>";
250 								strPageBtn += "<ul>";
251 								//首页 and 上一页
252 								if(pageNum<=1){
253 									strPageBtn += "<li class='pfirst disabled'><a href='#'><i class='icon-double-angle-left'></i></a></li>";
254 									strPageBtn += "<li class='prev disabled'><a href='#'><i class='icon-angle-left'></i></a></li>";
255 								}else{
256 									strPageBtn += "<li class='pfirst'><a href='#'><i class='icon-double-angle-left'></i></a></li>";
257 									strPageBtn += "<li class='prev'><a href='#'><i class='icon-angle-left'></i></a></li>";
258 								}
259 								//中间数字
260 								strPageBtn += strPageNum;
261 
262 								//下一页 and 尾页
263 								if(pageNum == pageCount){
264 									strPageBtn += "<li class='next disabled'><a href='#'><i class='icon-angle-right'></i></a></li>";
265 									strPageBtn += "<li class='plast disabled'><a href='#'><i class='icon-double-angle-right'></i></a></li>";
266 								}else{
267 									strPageBtn += "<li class='next'><a href='#'><i class='icon-angle-right'></i></a></li>";
268 									strPageBtn += "<li class='plast'><a href='#'><i class='icon-double-angle-right'></i></a></li>";
269 								}
270 
271 								//总计信息
272 								strPageBtn += "<li class='disabled'><a href='#'>共<b>"+pageCount+"</b>页"+dataCount+"条</a></li>";
273 								strPageBtn += "</ul>";
274 								strPageBtn += "</div>";
275 
276 								//插入到dom中显示出来
277 								_dom.find("#pagination").html(strPageBtn);
278 								
279 								//获取分页的dom
280 								$pageFirst = _dom.find("#pagination li.pfirst").not(".disabled");  //首页 Dom
281 								$pagePrev = _dom.find("#pagination li.prev").not(".disabled");     //上页 Dom
282 								$pageNum = _dom.find("#pagination li.pagenum").not(".active");     //中间数字 Dom
283 								$pageNext = _dom.find("#pagination li.next").not(".disabled");     //下页 Dom
284 								$pageLast = _dom.find("#pagination li.plast").not(".disabled");    //尾页 Dom
285 
286 								//给分页按钮绑定点击事件
287 								$pageFirst.on("click",function(){
288 									reShowFormList(1);
289 								})
290 								$pagePrev.on("click",function(){
291 									var curPageNum = parseInt(_dom.find("#pagination li.pagenum.active").text());
292 									reShowFormList(curPageNum-1);
293 								})
294 								$pageNum.on("click",function(){
295 									var curPageNum = parseInt($(this).text());
296 									reShowFormList(curPageNum);
297 								})
298 								$pageNext.on("click",function(){
299 									var curPageNum = parseInt(_dom.find("#pagination li.pagenum.active").text());
300 									reShowFormList(curPageNum+1);
301 								})
302 								$pageLast.on("click",function(){
303 									reShowFormList(pageCount);
304 								})
305 
306 								//回调函数
307 								_bindFormListCallBack();
308 							}
309 						}else if(code == 21){
310 							alert("未登录！");
311 						}else{
312 							alert(_this.alias+"列表内容请求报错！");
313 						}
314 					})
315 				}
316 
317 				//默认显示第一页
318 				reShowFormList(1);
319 
320 			}else{
321 				alert("列表分页总数获取失败！");
322 			}
323 		})
324 	}

326 	/**
327 	 * @function
328 	 * @memberOf lang
329 	 * @name lang
330 	 * @desctiption 使用语言信息格式化要显示的内容，如果要显示的内容对象包含了多语信息，则显示成多语言，否则直接显示成字符串
331 	 * @param {String}_displayObj 对象描述符 
332 	 */
333 	_result.lang = function(__displayObj){
334 		try{
335 			var langObj = eval("("+__displayObj+")");
336 			var lang = Cfg.lang;
337 			if (_SYS_PageLang){
338 				lang = _SYS_PageLang;
339 			}
340 			if (langObj[lang]){
341 				return langObj[lang];
342 			}
343 			return __displayObj;
344 		}catch(e){
345 			return __displayObj;
346 		}
347 	}
348 
349 	return _result;
